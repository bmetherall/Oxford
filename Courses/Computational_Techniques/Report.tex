\begin{theorem}[Eckart--Young--Mirsky]
The $r$-truncated SVD of a matrix, $A$ is the best rank $r$ approximation in the spectral norm.
\end{theorem}

\begin{proof}
this is the proof

\begin{align*}
\| A - B \|_2^2 &\geq \| \left( A - B \right) x \|_2^2 \\
&= \| Ax \|_2^2 \\
&= \| U \Sigma V^T x \|_2^2 \\
&= \| \Sigma V^T x \|_2^2 \\
&= \text{expand} \\
&\geq \sigma_{r+1}^2 \\
&= \| A - A_r \|_2^2
\end{align*}
\end{proof}
